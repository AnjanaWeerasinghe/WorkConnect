rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    // Users collection - users can read and write their own documents
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Workers collection - allow authenticated users to read worker profiles
    match /workers/{workerId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && request.auth.uid == workerId;
    }
    
    // Jobs collection
    match /jobs/{jobId} {
      allow read: if request.auth != null && 
        (request.auth.uid == resource.data.customerId || 
         request.auth.uid == resource.data.workerId);
      allow create: if request.auth != null && request.auth.uid == request.resource.data.customerId;
      allow update: if request.auth != null && 
        (request.auth.uid == resource.data.customerId || 
         request.auth.uid == resource.data.workerId);
    }
    
    // Reviews collection
    match /reviews/{reviewId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null && request.auth.uid == request.resource.data.customerId;
      allow update: if request.auth != null && request.auth.uid == resource.data.customerId;
    }
    
    // Messages collection (for chat functionality)
    match /messages/{messageId} {
      allow read, write: if request.auth != null && 
        (request.auth.uid == resource.data.senderId || 
         request.auth.uid == resource.data.receiverId);
    }
    
    // Allow authenticated users to read and write their own data during development
    // Remove or restrict this in production
    match /{document=**} {
      allow read, write: if request.auth != null;
    }
  }
}
